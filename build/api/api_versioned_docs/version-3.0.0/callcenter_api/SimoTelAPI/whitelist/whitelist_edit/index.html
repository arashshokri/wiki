نسخه: API v3

WhiteList Edit

از این سرویس برای ویرایش شماره‌های موجود در لیست سفید استفاده می‌شود.

پارامتر‌ها​

توضیحاتداده های تعریف شدهداده های نمونهپارامترهای ضروری[**]/منطقی[*]پارامترها
شناسه‌یکتای که از جستجو بدست می‌آید-6033876dc92de036d1390923**id_
شماره-555*number
--ext_test*labels
توضیحات--comment

نمونه خروجی درخواست​

{
    "success": 1,
    "message": "Requested operation is done successfully",
    "data": "",
    "old": null
}

  • JS

<?php

$curl = curl_init();

curl_setopt_array($curl, array(
  CURLOPT_URL => 'http://192.168.51.20/api/v3/pbx/whitelists',
  CURLOPT_RETURNTRANSFER => true,
  CURLOPT_ENCODING => '',
  CURLOPT_MAXREDIRS => 10,
  CURLOPT_TIMEOUT => 0,
  CURLOPT_FOLLOWLOCATION => true,
  CURLOPT_HTTP_VERSION => CURL_HTTP_VERSION_1_1,
  CURLOPT_CUSTOMREQUEST => 'PUT',
  CURLOPT_POSTFIELDS =>'{
    "_id":"6033876dc92de036d1390923",
    "number":"555",
    "labels":["ext_test"],
    "comment":""
}',
  CURLOPT_HTTPHEADER => array(
    'X-APIKEY: vZKtIKWsld0egNlkzHo8i5LVBqLNBSWARCQsPOSgDjFmAHM3tG',
    'Authorization: Basic dGVzdDpBYTEyMzQ1Ng==',
    'Content-Type: application/json'
  ),
));

$response = curl_exec($curl);


if (!curl_errno($curl)) {
    $httpcode = curl_getinfo($curl, CURLINFO_HTTP_CODE);
    echo 'response code:'.$httpcode, '<br/>';
}


curl_close($curl);
echo $response;
